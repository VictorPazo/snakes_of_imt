//Schema of DataBase

CREATE TABLE pessoa (
    id INT PRIMARY KEY AUTO_INCREMENT,
    nome VARCHAR(100) NOT NULL,
    email VARCHAR(150) UNIQUE NOT NULL,
    telefone VARCHAR(15),
    senha VARCHAR(255) NOT NULL
);

CREATE TABLE cobra (
    id INT PRIMARY KEY AUTO_INCREMENT,
    família VARCHAR (30) NOT NULL,
    gênero VARCHAR (30) NOT NULL,
    especie VARCHAR(50) NOT NULL,
    descricao TEXT,
    venenosa BOOLEAN NOT NULL
);

CREATE TABLE foto_cobra (
    id INT PRIMARY KEY AUTO_INCREMENT,
    cobra_id INT NOT NULL,
    pessoa_id INT NOT NULL,
    data_foto DATETIME DEFAULT CURRENT_TIMESTAMP,
    
    imagem LONGBLOB NOT NULL,

    FOREIGN KEY (cobra_id) REFERENCES cobra(id)
        ON DELETE CASCADE,
        
    FOREIGN KEY (pessoa_id) REFERENCES pessoa(id)
        ON DELETE CASCADE
);
